PREFIX musica: <http://www.semanticweb.org/lorenzo/ontologies/music#>

1)
SELECT ?album (COUNT (?brano) AS ?count)
	WHERE { ?brano musica:contenutoIn ?album }
GROUP BY ?album

2)
SELECT ?band ?membro
	WHERE { ?band musica:haComponente ?membro }
ORDER BY ?band

3)
SELECT ?album ?brano
	WHERE { ?brano musica:contenutoIn ?album }
ORDER BY ?album

4)
SELECT ?genere WHERE { ?genere a musica:GenereMusicale }

5)
SELECT ?canale ?brano WHERE
	 { ?brano musica:distribuitoTramite ?canale }
ORDER BY ?canale
